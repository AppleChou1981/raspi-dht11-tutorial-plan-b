1
如何正確複製 Google 試算表 ID？

請打開你的雲端試算表，觀察瀏覽器上方網址列，結構如下圖。請僅複製兩條斜線中間的那一長串隨機亂碼：

https://docs.google.com/spreadsheets/d/【 只要複製這整串英數亂碼，這就是試算表 ID 】/edit#gid=0

💡 註：這串 ID 稍後需要貼進 Python 程式碼中的 SPREADSHEET_ID 欄位。

2
硬體腳位對照表 (Raspberry Pi 3 40-Pin 排針)

請對照下表進行接線，並確認 DHT11 模組上的字樣標示（因市面上排針順序不同，務必以模組上的文字標記為準）：

左側功能標記 (奇數針腳) 實體 實體 右側功能標記 (偶數針腳)
接 DHT11 的 VCC (+) ── [3.3V 電源] 1 2 5V 電源
GPIO 2 (SDA) 3 4 5V 電源
GPIO 3 (SCL) 5 6 接 DHT11 的 GND (-) ── [Ground 地線]
接 DHT11 的 DAT (Data/Out/S) ── [GPIO 4] 7 8 GPIO 14 (TXD)
Ground (接地) 9 10 GPIO 15 (RXD)
(以下細節省略，上方已涵蓋本次實驗所需全部腳位)
3
樹莓派本地中繼控制主程式

在樹莓派中建立 dht11_local_relay.py，並將以下代碼完整貼入：

import time
import requests
import board
import adafruit_dht
import gspread
from oauth2client.service_account import ServiceAccountCredentials

# ==================== 1. 雲端與金鑰引腳參數設定 ====================

JSON_KEY_FILE = "creds.json"  # 從 GCP 下載的金鑰檔案名

# 📢 在這裡貼上你在步驟 1 所複製的長串試算表 ID
SPREADSHEET_ID = "請在此處替換為你的_Google_試算表網址中的長串ID"

LINE_ACCESS_TOKEN = "請在此處貼上你的_LINE_Channel_Access_Token"
TARGET_USER_ID = "請在此處貼上你的_LINE_Your_user_ID"

# 預設連接到 GPIO 4 (即實體第 7 針腳)
dhtDevice = adafruit_dht.DHT11(board.D4)

# ==================== 2. 核心功能邏輯 ====================

def send_to_google_sheet_direct(temp, humi):
    scope = ["https://spreadsheets.google.com/feeds", "https://www.googleapis.com/auth/drive"]
    try:
        creds = ServiceAccountCredentials.from_json_keyfile_name(JSON_KEY_FILE, scope)
        client = gspread.authorize(creds)
        sheet = client.open_by_key(SPREADSHEET_ID).sheet1
        
        now = time.strftime("%Y-%m-%d %H:%M:%S", time.localtime())
        sheet.append_row([now, temp, humi])
        print("【Google Sheets】雲端試算表安全寫入成功！")
    except Exception as e:
        print(f"【Google Sheets】寫入錯誤: {e}")

def send_to_line_bot_direct(temp, humi):
    url = "https://api.line.me/v2/bot/message/push"
    current_time = time.strftime("%Y-%m-%d %H:%M:%S", time.localtime())
    message_text = f"\n🌡️ 樹莓派回報 (方案B)\n-------------------\n⏰ 時間：{current_time}\n°C 溫度：{temp} °C\n💧 濕度：{humi} %"
    
    headers = {
        "Content-Type": "application/json",
        "Authorization": f"Bearer {LINE_ACCESS_TOKEN}"
    }
    payload = {
        "to": TARGET_USER_ID,
        "messages": [{"type": "text", "text": message_text}]
    }
    try:
        response = requests.post(url, json=payload, headers=headers, timeout=10)
        if response.status_code == 200:
            print("【LINE Bot】即時訊息推播成功！")
        else:
            print(f"【LINE Bot】發送失敗: {response.status_code}")
    except Exception as e:
        print(f"【LINE Bot】連線異常: {e}")

# ==================== 3. 主程式執行無窮迴圈 ====================
if __name__ == "__main__":
    print("樹莓派 方案 B 系統啟動，每 10 分鐘自動讀取寫入...")
    try:
        while True:
            try:
                temperature = dhtDevice.temperature
                humidity = dhtDevice.humidity
                
                if humidity is not None and temperature is not None:
                    send_to_google_sheet_direct(round(temperature, 1), round(humidity, 1))
                    send_to_line_bot_direct(round(temperature, 1), round(humidity, 1))
                time.sleep(600)
            except RuntimeError as error:
                time.sleep(2.0)
                continue
            except Exception as error:
                dhtDevice.exit()
                raise error
    except KeyboardInterrupt:
        dhtDevice.exit()
        print("\n程式已手動終止。")
